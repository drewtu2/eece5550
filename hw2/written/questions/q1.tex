\subsection{Rotation Matrix}
$$ R_{robot}^{world} = 
\begin{bmatrix} 
cos\theta & -sin\theta & 0 \\
sin\theta & cos\theta & 0 \\
0 & 0 & 1 \\
\end{bmatrix}
$$

$$ v_{world} = R_{robot}^{world}* 
\begin{bmatrix} 
vx _{robot}\\
vy_{robot} \\
v\theta_{robot} \\
\end{bmatrix}
$$
